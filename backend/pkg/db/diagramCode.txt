Table users {
    id uuid [primary key]
    email text 
    password text 
    first_name text 
    last_name text 
    date_of_birth date
    avatar_path text
    nickname text
    about_me text
    is_public bool
    created_at timestamp
}

Table followers{
  follower_id uuid
  following_id uuid
  status text
}

Table posts{
  id uuid
  user_id uuid
  content text
  image_path text
  privacy text
  created_at timestamp
}

Table comments{
  id uuid
  post_id uuid
  user_id uuid
  content text
  created_at timestamp
}

Table groups{
  id uuid
  title text
  description text
  creator_id uuid
  is_public bool
  created_at timestamp
}

Table groupMembers{
  group_id uuid
  user_id uuid
  status text
}

Table groupEvents{
  id uuid 
  group_id uuid
  title text
  description text
  day_time timestamp
}

Table eventParticipants{
  event_id uuid
  user_id uuid
  response text
}

Table privateMessages{
  id uuid
  sender_id uuid
  recipient uuid 
  content text
  created_at timestamp
}

Table notifications{
  id uuid
  user_id uuid
  content text
  type text
  is_read bool
  created_at timestamp
}
